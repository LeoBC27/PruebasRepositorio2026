<Programa>::= <Bloque> .

<Bloque>::=<LinealC><LinealV><LinealP><Proposicion>
<LinealC>::= <LinealConst>|∅
<LinealConst>::= const<CicloNum>;
<CicloNum>::= id=num<CicloNumPrima>
<CicloNumPrima>::= ,<CicloNum>|∅
<LinealV>::= <LinealVar>|∅
<LinealVar>::= var<CicloId>;
<CicloId>::= id<CicloIdPrima>
<CicloIdPrima>::= ,<CicloId>|∅
<LinealP>::=<CicloProc>|∅
<CicloProc>::= <LinealProc><CicloProcPrima>
<CicloProcPrima>::= <CicloProc>|∅
<LinealProc>::= proced id;<Bloque>;


<Condicion>::= <Expresion><MultSigno><Expresion>
<MultSigno>::= ==|!=|<|>|<=|>=

<Proposicion>::= <MultPropo>
<MultPropo>::= <LinealBegin>|<LinealId>|<LinealWrite>|<LinealRead>|<LinealCall>|<LinealIf>|<LinealWhile>|<LinealFor>
<LinealBegin>::= begin<CicloProp>end
<CicloProp>::= <Proposicion><CicloPropPrima>
<CicloPropPrima>::= ;<CicloProp>|∅
<LinealId>::= id=<Expresion>
<LinealWrite>::= write<DisyWri>
<DisyWri>::= id|num
<LinealRead>::= read id
<LinealCall>::= call id
<LinealIf>::= if<Condicion> then <Proposicion>
<LinealWhile>::= while<Condicion> do <Proposicion>
<LinealFor>::= for id =<Expresion><DisyExp><Expresion> do <Proposicion>
<DisyExp>::= to|down

<Expresion>::= <Termino><ExpresionPrima>
<ExpresionPrima>::= <CicloTerm><Termino><ExpresionPrima>|∅
<CicloTerm>::= <DisySig>
<DisySig>::= +|-

<Termino>::= <Factor><TerminoPrima>
<TerminoPrima>::= <CicloFact><Factor><TerminoPrima>|∅
<CicloFact>::= <DisySigno>
<DisySigno>::= *|/

<Factor>::=<MultExp>
<MultExp>::= (<Expresion>)|id|num
