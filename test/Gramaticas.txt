<Programa> -> <Bloque> .

<Bloque> -> <LinealC> <LinealV> <LinealP> <Proposicion>
<LinealC> -> const <CicloNum> ;
<LinealC> -> ∅
<CicloNum> -> id = num
<CicloNum> -> id = num , <CicloNum>
<LinealV> -> var <CicloId> ;
<LinealV> -> ∅
<CicloId> -> id
<CicloId> -> id , <CicloId>
<LinealP> -> proced id ; <Bloque> ; <LinealP>
<LinealP> -> ∅

<Condicion> -> <Expresion> <MultSigno> <Expresion>
<MultSigno> -> ==
<MultSigno> -> !=
<MultSigno> -> <
<MultSigno> -> >
<MultSigno> -> <=
<MultSigno> -> >=

<Proposicion> -> <LinealBegin>
<Proposicion> -> <LinealId>
<Proposicion> -> <LinealWrite>
<Proposicion> -> <LinealRead>
<Proposicion> -> <LinealCall>
<Proposicion> -> <LinealIf>
<Proposicion> -> <LinealWhile>
<Proposicion> -> <LinealFor>
<LinealBegin> -> begin <CicloProp> end
<CicloProp> -> <Proposicion>
<CicloProp> -> <Proposicion> ; <CicloProp>
<LinealId> -> id = <Expresion>
<LinealWrite> -> write <DisyWri>
<DisyWri> -> id
<DisyWri> -> num
<LinealRead> -> read id
<LinealCall> -> call id
<LinealIf> -> if <Condicion> then <Proposicion>
<LinealWhile> -> while <Condicion> do <Proposicion>
<LinealFor> -> for id = <Expresion> <DisyExp> <Expresion> do <Proposicion>
<DisyExp> -> to
<DisyExp> -> down

<Expresion> -> <Termino>
<Expresion> -> <Termino> <DisySig> <Expresion>
<DisySig> -> +
<DisySig> -> -

<Termino> -> <Factor>
<Termino> -> <Factor> <DisySigno> <Termino>
<DisySigno> -> *
<DisySigno> -> /

<Factor> -> (<Expresion>)
<Factor> -> id
<Factor> -> num
